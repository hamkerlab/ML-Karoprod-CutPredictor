    from cut_predictor import CutPredictor

reg = CutPredictor(
    data = data,
    process_parameters = [
        'Material_ID', 
        'Niederhalterkraft', 
        'Stempel_ID', 
        'Einlegeposition', 
        'Ziehtiefe'
    ],
    categorical = [
        'Material_ID', 
        'Stempel_ID', 
        'Einlegeposition', 
        'Ziehtiefe'
    ],
    position = 'c_phi',
    output = 'c_z',
    angle=True,
)
reg.data_summary()
