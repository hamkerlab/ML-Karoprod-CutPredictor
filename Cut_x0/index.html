    Creating the regressor#

    from cut_predictor import CutPredictor

reg = CutPredictor()
reg.load_data(
    doe = doe,
    data = data,
    index='doe_id',
    process_parameters = [
        'Blechdicke', 
        'Niederhalterkraft', 
        'Ziehspalt', 
        'Einlegeposition', 
        'Ziehtiefe',
        'Stempel_ID',
        'E',
        'Rp0',
        'Rp50'
    ],
    categorical = [
        'Ziehspalt', 
        'Ziehtiefe',
        'Stempel_ID',
    ],
    position = 'tp',
    output = ['deviationc', 'y', 'z']
)

    # Print a summary of the data
reg.data_summary()

    Alternative: define a custom network and do the optimization yourself#

    One can also run the autotuning for a limited number of epochs and then fine-tune the best configuration by training it longer.

    config = {
    'batch_size': 4096,
    'max_epochs': 50,
    'layers': [128, 128, 128, 128, 128],
    'dropout': 0.0,
    'learning_rate': 0.001
}

# or best_config from autotune if you already did it once

reg.custom_model(save_path='best_x0_model', config=config, verbose=True)
reg.training_summary()
